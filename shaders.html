Basics Color 
				
Shader "LearnSurface/Color"
{
    Properties
    {
        _Color ("Color", Color) = (1,1,1,1)
        _MainTex ("Albedo (RGB)", 2D) = "white" {}
        _Glossiness ("Smoothness", Range(0, 1)) = 0.5
        _Metallic ("Metallic", Range(0, 1)) = 0.0
    }
    SubShader
    {
        Tags { "RenderType"="Opaque" }

        CGPROGRAM
        #pragma surface surf Standard fullforwardshadows vertex:vert
        #pragma target 5.0
        
        struct Input
        {
            float2 texcoord;
        };

        sampler2D _MainTex;
        float _Glossiness;
        float _Metallic;
        float4 _Color;
        
        void vert(inout appdata_full app, out Input input)
        {
            UNITY_INITIALIZE_OUTPUT(Input, input);
            input.texcoord = app.texcoord.xy;
        }

        void surf(Input input, inout SurfaceOutputStandard o)
        {
            o.Albedo = tex2D(_MainTex, input.texcoord).rgb * _Color.rgb;
            o.Metallic = _Metallic;
            o.Smoothness = _Glossiness;
        }
        ENDCG
    }
    FallBack "Diffuse"
}
